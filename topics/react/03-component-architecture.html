Concept 1

Functional Components — the only component type you need

Prior to hooks (React 16.8), you needed class components for state and lifecycle. Today, functional components handle everything — and they are simpler to read, test, and compose.

// Modern functional component — the standard
function UserCard({ user, onEdit }) {
  const [expanded, setExpanded] = useState(false);

  return (
    <article className="user-card">
      <header>
        <img src={user.avatarUrl} alt={user.name} />
        <h2>{user.name}</h2>
      </header>

      {expanded && <p className="bio">{user.bio}</p>}

      <footer>
        <button onClick={() => setExpanded(e => !e)}>
          {expanded ? 'Less' : 'More'}
        </button>
        <button onClick={() => onEdit(user.id)}>Edit</button>
      </footer>
    </article>
  );
}

// Compare with old class component — same behaviour, much more boilerplate
class UserCardClass extends React.Component {
  state = { expanded: false };

  render() {
    const { user, onEdit } = this.props;
    const { expanded } = this.state;

    return (
      <article>
        {/* same JSX */}
        <button onClick={() => this.setState({ expanded: !expanded })}>
          Toggle
        </button>
      </article>
    );
  }
}

Concept 2

Reusable Components — building a component library

A reusable component does one thing well, accepts the right props to cover its use cases, and doesn't assume where it will be used. The test: can you use this component in two completely different contexts without changing it?

// ─── Reusable Button ─────────────────────────────────────────────────────
// Covers: variants, sizes, loading state, icon, disabled, all native button props
function Button({
  children,
  variant = 'primary',   // primary | secondary | ghost | danger
  size = 'md',           // sm | md | lg
  loading = false,
  leftIcon,
  rightIcon,
  className,
  ...rest                // passes through type, onClick, disabled, aria-*, etc.
}) {
  return (
    <button
      className={['btn', `btn--${variant}`, `btn--${size}`, className].filter(Boolean).join(' ')}
      disabled={loading || rest.disabled}
      {...rest}
    >
      {loading && <Spinner size={16} />}
      {!loading && leftIcon}
      {children}
      {!loading && rightIcon}
    </button>
  );
}

// Usage in many different contexts:
<Button variant="primary" onClick={handleSubmit}>Save Changes</Button>
<Button variant="danger" size="sm" onClick={handleDelete}>Delete</Button>
<Button variant="ghost" loading={isSaving}>Saving...</Button>
<Button variant="secondary" leftIcon={<DownloadIcon />}>Export CSV</Button>

// ─── Reusable Input ───────────────────────────────────────────────────────
function Input({
  label,
  error,
  helperText,
  id,
  required,
  className,
  ...rest
}) {
  const inputId = id ?? `input-${label?.toLowerCase().replace(/\s/g, '-')}`;

  return (
    <div className={`form-field ${error ? 'form-field--error' : ''} ${className ?? ''}`}>
      {label && (
        <label htmlFor={inputId}>
          {label} {required && <span aria-hidden>*</span>}
        </label>
      )}
      <input id={inputId} aria-describedby={`${inputId}-helper`} {...rest} />
      <span id={`${inputId}-helper`} className={error ? 'error-text' : 'helper-text'}>
        {error ?? helperText}
      </span>
    </div>
  );
}

// ─── Reusable Badge ───────────────────────────────────────────────────────
const BADGE_COLORS = {
  green: 'bg-green-100 text-green-800',
  red:   'bg-red-100 text-red-800',
  blue:  'bg-blue-100 text-blue-800',
  gray:  'bg-gray-100 text-gray-700',
};

function Badge({ children, color = 'gray' }) {
  return (
    <span className={`badge ${BADGE_COLORS[color]}`}>{children}</span>
  );
}
Concept 3

Composition — building complex UI from simple parts

Composition means assembling complex behavior by combining small, focused components. This is React's answer to inheritance in OOP — instead of extending components, you wrap them.

// ─── Composition vs inheritance ───────────────────────────────────────────
// Bad: inheriting from another component (React doesn't recommend this)
class SpecialButton extends Button { /* can't easily do this cleanly */ }

// Good: composing components
function IconButton({ icon, label, ...props }) {
  return (
    <Button {...props}>
      {icon}
      <span>{label}</span>
    </Button>
  );
}

// ─── Slot-based composition (common in UI libraries) ─────────────────────
// Card with named slots for header, body, footer
function Card({ header, footer, children, className }) {
  return (
    <div className={`card ${className ?? ''}`}>
      {header && <div className="card__header">{header}</div>}
      <div className="card__body">{children}</div>
      {footer && <div className="card__footer">{footer}</div>}
    </div>
  );
}

// Usage — caller controls what goes in each slot
<Card
  header={<h2>User Settings</h2>}
  footer={<Button onClick={save}>Save</Button>}
>
  <SettingsForm user={user} />
</Card>

// ─── Containment + specialization ────────────────────────────────────────
// Base Dialog component
function Dialog({ title, children, actions }) {
  return (
    <div role="dialog" aria-modal="true" className="dialog">
      <h2>{title}</h2>
      <div className="dialog__content">{children}</div>
      <div className="dialog__actions">{actions}</div>
    </div>
  );
}

// Specialized confirmation dialog built by composing Dialog
function ConfirmDialog({ title, message, onConfirm, onCancel }) {
  return (
    <Dialog
      title={title}
      actions={
        <>
          <Button variant="ghost" onClick={onCancel}>Cancel</Button>
          <Button variant="danger" onClick={onConfirm}>Confirm</Button>
        </>
      }
    >
      <p>{message}</p>
    </Dialog>
  );
}
Concept 4

Children Props — the most powerful prop

children is a special prop that contains whatever you put between a component's opening and closing tags. It enables the most flexible composition patterns in React.

// children is the JSX between opening and closing tags
function Panel({ children, title }) {
  return (
    <div className="panel">
      <h3>{title}</h3>
      <div className="panel__body">{children}</div>
    </div>
  );
}

// Usage — anything between the tags becomes children
<Panel title="Settings">
  <Input label="Name" />
  <Input label="Email" />
  <Button>Save</Button>
</Panel>

// ─── children as a function (render props) ─────────────────────────────
// The child is a function that receives data from the parent
function Toggle({ children }) {
  const [on, setOn] = useState(false);
  return children({ on, toggle: () => setOn(o => !o) });
}

// Usage — caller controls the rendering completely
<Toggle>
  {({ on, toggle }) => (
    <div>
      <p>The toggle is {on ? 'on' : 'off'}</p>
      <button onClick={toggle}>Toggle</button>
    </div>
  )}
</Toggle>

// ─── Inspecting children ──────────────────────────────────────────────────
// React.Children utilities for working with children programmatically
function Tabs({ children }) {
  const [active, setActive] = useState(0);

  const tabs = React.Children.map(children, (child, i) =>
    React.cloneElement(child, { isActive: i === active, onSelect: () => setActive(i) })
  );

  return <div className="tabs">{tabs}</div>;
}

// ─── Guarding against unexpected children ─────────────────────────────────
// React.Children.count for conditional rendering
function Grid({ children, minCols = 2 }) {
  const count = React.Children.count(children);
  const cols = Math.min(count, minCols);
  return (
    <div style={{ gridTemplateColumns: `repeat(${cols}, 1fr)` }} className="grid">
      {children}
    </div>
  );
}
Concept 5

Layout Components — shells for consistent page structure

Layout components control the structural skeleton of a page — the header, sidebar, footer, and content area — without caring about what content goes inside. They are essential for consistent spacing and navigation across pages.

// ─── App shell layout ─────────────────────────────────────────────────────
function AppShell({ children }) {
  return (
    <div className="app-shell">
      <TopNav />
      <div className="app-shell__body">
        <Sidebar />
        <main className="app-shell__content">
          {children}
        </main>
      </div>
    </div>
  );
}

// ─── Page layout with header and actions ─────────────────────────────────
function PageLayout({ title, breadcrumbs, actions, children }) {
  return (
    <div className="page">
      <header className="page-header">
        {breadcrumbs && <Breadcrumbs items={breadcrumbs} />}
        <div className="page-header__row">
          <h1>{title}</h1>
          {actions && <div className="page-header__actions">{actions}</div>}
        </div>
      </header>
      <div className="page-body">{children}</div>
    </div>
  );
}

// Usage
<AppShell>
  <PageLayout
    title="Users"
    breadcrumbs={[{ label: 'Home', href: '/' }, { label: 'Users' }]}
    actions={<Button onClick={openInviteModal}>Invite User</Button>}
  >
    <UserTable users={users} />
  </PageLayout>
</AppShell>

Concept 6

Design Patterns — proven solutions to common React problems

// ─── Pattern 1: Container / Presenter (Smart / Dumb) ────────────────────
// Container handles data fetching and state
function UserListContainer() {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetchUsers().then(setUsers).finally(() => setLoading(false));
  }, []);

  if (loading) return <Spinner />;
  return <UserList users={users} />;  // pass data to presenter
}

// Presenter handles only rendering — no data logic, easy to test
function UserList({ users }) {
  return (
    <ul>
      {users.map(u => <li key={u.id}>{u.name}</li>)}
    </ul>
  );
}

// ─── Pattern 2: Compound Components ──────────────────────────────────────
// Related components that share implicit state via context
// e.g. <Select> with <Select.Option> children
const SelectContext = React.createContext(null);

function Select({ value, onChange, children }) {
  return (
    <SelectContext.Provider value={{ value, onChange }}>
      <div className="select">{children}</div>
    </SelectContext.Provider>
  );
}

function Option({ value, children }) {
  const { value: selected, onChange } = useContext(SelectContext);
  return (
    <div
      className={`option ${selected === value ? 'option--selected' : ''}`}
      onClick={() => onChange(value)}
    >
      {children}
    </div>
  );
}

Select.Option = Option;  // Attach as static property

// Usage — clean, self-documenting API
<Select value={role} onChange={setRole}>
  <Select.Option value="admin">Admin</Select.Option>
  <Select.Option value="editor">Editor</Select.Option>
  <Select.Option value="viewer">Viewer</Select.Option>
</Select>

// ─── Pattern 3: HOC — Higher Order Component ─────────────────────────────
// A function that takes a component and returns an enhanced component
// Common in auth guards, analytics, logging
function withAuth(WrappedComponent) {
  return function AuthenticatedComponent(props) {
    const { user, loading } = useAuth();
    if (loading) return <Spinner />;
    if (!user)   return <Navigate to="/login" />;
    return <WrappedComponent {...props} user={user} />;
  };
}
const ProtectedDashboard = withAuth(Dashboard);

// ─── Pattern 4: Custom Hook (preferred over HOC in modern React) ──────────
// Extracts logic into a reusable function — simpler than HOC
function useAuth() {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    getCurrentUser().then(setUser).finally(() => setLoading(false));
  }, []);

  return { user, loading, isAuthenticated: !!user };
}

// Usage — cleaner than HOC, works with any component
function Dashboard() {
  const { user, loading } = useAuth();
  if (loading) return <Spinner />;
  return <h1>Welcome, {user.name}</h1>;
}
Prefer custom hooks over HOCs in modern React. HOCs wrap your component tree (harder to debug), can cause props name collisions, and are harder to type in TypeScript. Custom hooks extract logic while keeping your component tree flat. HOCs still appear in older codebases — know both.

Chapter 3 Takeaway

Component architecture is fundamentally about separation of concerns. Layout components own structure. Container components own data. Presenter components own appearance. Compound components provide ergonomic APIs. The goal is small, focused components that can be combined without fighting each other. When you find yourself copy-pasting JSX, that is the signal to extract a reusable component.
