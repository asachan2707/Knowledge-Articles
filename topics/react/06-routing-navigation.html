import { BrowserRouter, Routes, Route, Link, NavLink } from 'react-router-dom';

function App() {
  return (
    <BrowserRouter>
      <nav>
        <Link to="/">Home</Link>
        {/* NavLink adds aria-current and active class automatically */}
        <NavLink to="/products" className={({ isActive }) => isActive ? 'nav-active' : ''}>

        </NavLink>
      </nav>

      <Routes>
        <Route path="/"         element={<HomePage />} />
        <Route path="/products" element={<ProductsPage />} />
        <Route path="/about"    element={<AboutPage />} />
        <Route path="*"         element={<NotFoundPage />} />
      </Routes>
    </BrowserRouter>

// Reading URL params and search params
import { useParams, useSearchParams, useLocation, useNavigate } from 'react-router-dom';

function ProductDetailPage() {
  const { productId } = useParams();              // /products/:productId
  const [searchParams] = useSearchParams();       // ?tab=reviews&page=2
  const location  = useLocation();                // { pathname, search, hash, state }
  const navigate  = useNavigate();                // programmatic navigation

  const tab  = searchParams.get('tab') ?? 'details';
  const page = Number(searchParams.get('page') ?? 1);

  return (
    <div>
      <button onClick={() => navigate(-1)}>Back</button>
      <button onClick={() => navigate('/products', { replace: true })}>
        All Products
      </button>
      <ProductDetail id={productId} tab={tab} page={page} />
    </div>
  );
}
Concept 2

Nested Routes — layouts that persist across pages

Nested routes let a parent layout render persistently while child routes swap in and out via <Outlet />. This is how dashboards and tabbed pages work — the shell stays, the content changes.

import { Outlet } from 'react-router-dom';

// Dashboard layout — renders once, persists while child routes change
function DashboardLayout() {
  return (
    <div className="dashboard">
      <DashboardSidebar />
      <main>
        <Outlet />  {/* child route renders here */}
      </main>
    </div>
  );
}

// Route config — nested structure
function App() {
  return (
    <BrowserRouter>
      <Routes>
        {/* Public routes */}
        <Route path="/"      element={<LandingPage />} />
        <Route path="/login" element={<LoginPage />} />

        {/* Dashboard — all children render inside DashboardLayout's Outlet */}
        <Route path="/dashboard" element={<DashboardLayout />}>
          <Route index element={<DashboardHome />} />        {/* /dashboard */}
          <Route path="users"    element={<UsersPage />} />   {/* /dashboard/users */}
          <Route path="orders"   element={<OrdersPage />} />  {/* /dashboard/orders */}
          <Route path="settings" element={<SettingsPage />} />{/* /dashboard/settings */}

          {/* Nested further — user detail with tabs */}
          <Route path="users/:userId" element={<UserDetailLayout />}>
            <Route index           element={<UserProfileTab />} />
            <Route path="activity" element={<UserActivityTab />} />
            <Route path="orders"   element={<UserOrdersTab />} />
          </Route>
        </Route>

        <Route path="*" element={<NotFound />} />
      </Routes>
    </BrowserRouter>
  );
}
Concept 3

Protected Routes — auth guards

import { Navigate, Outlet, useLocation } from 'react-router-dom';
import { useAuth } from '../hooks/useAuth';

// ─── Auth guard component ─────────────────────────────────────────────────
function RequireAuth({ allowedRoles }) {
  const { user, loading } = useAuth();
  const location = useLocation();

  if (loading) return <FullPageSpinner />;

  if (!user) {
    // Redirect to login, save the attempted URL for redirect-back after login
    return <Navigate to="/login" state={{ from: location }} replace />;
  }

  if (allowedRoles && !allowedRoles.some(role => user.roles.includes(role))) {
    return <Navigate to="/unauthorized" replace />;
  }

  return <Outlet />;
}

// ─── Usage in route config ────────────────────────────────────────────────
<Routes>
  {/* Public */}
  <Route path="/" element={<LandingPage />} />
  <Route path="/login" element={<LoginPage />} />

  {/* Protected: any logged-in user */}
  <Route element={<RequireAuth />}>
    <Route path="/dashboard" element={<DashboardLayout />}>
      <Route index element={<DashboardHome />} />
    </Route>
  </Route>

  {/* Admin only */}
  <Route element={<RequireAuth allowedRoles={['admin']} />}>
    <Route path="/admin" element={<AdminPanel />} />
  </Route>
</Routes>

// ─── Login page: redirect back after login ────────────────────────────────
function LoginPage() {
  const navigate  = useNavigate();
  const location  = useLocation();
  const { login } = useAuth();
  const from = location.state?.from?.pathname ?? '/dashboard';

  const handleLogin = async (credentials) => {
    await login(credentials);
    navigate(from, { replace: true });
  };

  return <LoginForm onSubmit={handleLogin} />;
}

Dynamic Routes — URL parameters

// Route definition
<Route path="/products/:categorySlug/:productId" element={<ProductDetail />} />

// Accessing params
function ProductDetail() {
  const { categorySlug, productId } = useParams();

  // Use productId to fetch data
  const { data: product, isLoading } = useQuery({
    queryKey: ['product', productId],
    queryFn: () => fetchProduct(productId),
  });

  if (isLoading) return <Skeleton />;
  return <ProductView product={product} />;
}

// ─── Search params for filters/pagination ─────────────────────────────────
function ProductsPage() {
  const [searchParams, setSearchParams] = useSearchParams();

  const category = searchParams.get('category') ?? 'all';
  const sort     = searchParams.get('sort') ?? 'newest';
  const page     = Number(searchParams.get('page') ?? 1);

  const updateFilter = (key, value) => {
    setSearchParams(prev => {
      const next = new URLSearchParams(prev);
      next.set(key, value);
      next.set('page', '1');  // reset page on filter change
      return next;
    });
  };

  return (
    <div>
      <select value={category} onChange={e => updateFilter('category', e.target.value)}>
        <option value="all">All</option>
        <option value="electronics">Electronics</option>
      </select>
      <ProductGrid category={category} sort={sort} page={page} />
    </div>
  );
}
Concept 5

Route Loaders — fetch data before rendering

React Router 6.4+ loaders let you fetch data as a side effect of navigation — before the component renders. This eliminates the loading spinner flash on initial page load.

// Using createBrowserRouter with loaders
import { createBrowserRouter, RouterProvider, useLoaderData } from 'react-router-dom';

// Loader runs during navigation — before component renders
async function productLoader({ params }) {
  const [product, reviews] = await Promise.all([
    fetch(`/api/products/${params.productId}`).then(r => r.json()),
    fetch(`/api/products/${params.productId}/reviews`).then(r => r.json()),
  ]);
  return { product, reviews };
}

// Component receives loader data via useLoaderData (no loading state needed!)
function ProductDetailPage() {
  const { product, reviews } = useLoaderData();

  return (
    <div>
      <h1>{product.name}</h1>
      <ReviewList reviews={reviews} />
    </div>
  );
}

// Route config with loaders
const router = createBrowserRouter([
  {
    path: '/',
    element: <RootLayout />,
    children: [
      {
        path: 'products',
        element: <ProductsPage />,
        loader: async () => fetch('/api/products').then(r => r.json()),
      },
      {
        path: 'products/:productId',
        element: <ProductDetailPage />,
        loader: productLoader,
        errorElement: <ProductError />,  // shown if loader throws
      },
    ],
  },
]);

function App() {
  return <RouterProvider router={router} />;
}
Concept 6

Navigation Patterns — breadcrumbs, tabs, and programmatic routing

// ─── Breadcrumbs — built from current location ────────────────────────────
function Breadcrumbs() {
  const location = useLocation();
  const paths = location.pathname.split('/').filter(Boolean);

  const crumbs = paths.map((segment, i) => ({
    label: segment.replace(/-/g, ' '),
    href: '/' + paths.slice(0, i + 1).join('/'),
  }));

  return (
    <nav aria-label="breadcrumb">
      <ol>
        <li><Link to="/">Home</Link></li>
        {crumbs.map((crumb, i) => (
          <li key={crumb.href}>
            {i === crumbs.length - 1
              ? <span aria-current="page">{crumb.label}</span>
              : <Link to={crumb.href}>{crumb.label}</Link>
            }
          </li>
        ))}
      </ol>
    </nav>
  );
}

// ─── Tab navigation using URL search params ───────────────────────────────
const TABS = ['details', 'reviews', 'shipping'];

function ProductTabs() {
  const [searchParams, setSearchParams] = useSearchParams();
  const active = searchParams.get('tab') ?? 'details';

  return (
    <div>
      <div role="tablist">
        {TABS.map(tab => (
          <button
            key={tab}
            role="tab"
            aria-selected={active === tab}
            onClick={() => setSearchParams({ tab })}
          >
            {tab}
          </button>
        ))}
      </div>
      <div role="tabpanel">
        {active === 'details'  && <ProductDetails />}
        {active === 'reviews'  && <ProductReviews />}
        {active === 'shipping' && <ShippingInfo />}
      </div>
    </div>
  );
}

// ─── Programmatic navigation patterns ─────────────────────────────────────
function OrderForm() {
  const navigate = useNavigate();

  const handleSubmit = async (formData) => {
    const order = await createOrder(formData);
    // Navigate with state (accessible via useLocation in the target page)
    navigate(`/orders/${order.id}`, {
      state: { justCreated: true, fromCheckout: true },
    });
  };

  return <form onSubmit={handleSubmit}>{/* ... */}</form>;
}

// Block navigation when there are unsaved changes
import { useBlocker } from 'react-router-dom';

function EditForm({ isDirty }) {
  const blocker = useBlocker(() => isDirty);

  return (
    <div>
      {/* form */}
      {blocker.state === 'blocked' && (
        <ConfirmDialog
          message="You have unsaved changes. Leave anyway?"
          onConfirm={() => blocker.proceed()}
          onCancel={() => blocker.reset()}
        />
      )}
    </div>
  );
}
URL as state: Filters, pagination, and active tab should live in the URL, not in component state. URL state is free to share, bookmark, and use the back button with. Component state is lost on refresh. If users would be annoyed that the URL doesn't reflect their current view, put it in the URL.

Chapter 6 Takeaway

React Router v6 is built around three ideas: nested routes with Outlet for persistent layouts, URL params for dynamic segments, and loaders for data-before-render. Protected routes are just wrapper components that check auth before rendering Outlet. Move filter/pagination/tab state into URL search params — it makes apps shareable and back-button-friendly with minimal extra code.
