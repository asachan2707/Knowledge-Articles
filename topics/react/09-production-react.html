Concept 1

TypeScript — type-safe React components

// ─── Typing props ─────────────────────────────────────────────────────────
interface ButtonProps {
  children: React.ReactNode;
  variant?: 'primary' | 'secondary' | 'danger' | 'ghost';
  size?: 'sm' | 'md' | 'lg';
  loading?: boolean;
  onClick?: (event: React.MouseEvent<HTMLButtonElement>) => void;
  disabled?: boolean;
}

function Button({ children, variant = 'primary', size = 'md', loading = false, ...rest }: ButtonProps) {
  return <button className={`btn btn--${variant} btn--${size}`} {...rest}>{loading ? '...' : children}</button>;
}

// ─── Generic components ───────────────────────────────────────────────────
interface ListProps<T> {
  items: T[];
  renderItem: (item: T, index: number) => React.ReactNode;
  keyExtractor: (item: T) => string | number;
  emptyState?: React.ReactNode;
}

function List<T>({ items, renderItem, keyExtractor, emptyState }: ListProps<T>) {
  if (items.length === 0) return <>{emptyState}</>;
  return (
    <ul>
      {items.map((item, i) => (
        <li key={keyExtractor(item)}>{renderItem(item, i)}</li>
      ))}
    </ul>
  );
}

// Usage — fully typed
<List
  items={users}
  keyExtractor={u => u.id}
  renderItem={u => <UserRow user={u} />}
/>

// ─── Typing hooks ─────────────────────────────────────────────────────────
interface User { id: string; name: string; email: string; }

function useUser(userId: string) {
  const [user, setUser] = useState<User | null>(null);
  const [loading, setLoading] = useState(true);

// ─── Type narrowing ───────────────────────────────────────────────────────
type ApiResult<T> =
  | { status: 'loading' }
  | { status: 'error'; error: string }
  | { status: 'success'; data: T };

function UserView({ result }: { result: ApiResult<User> }) {
  if (result.status === 'loading') return <Spinner />;
  if (result.status === 'error')   return <Error message={result.error} />;
  // TypeScript knows result.data is User here
  return <h1>{result.data.name}</h1>;
}

Vitest — fast unit testing with Vite's infrastructure

// npm install -D vitest @vitest/ui
// vite.config.ts
import { defineConfig } from 'vitest/config';
export default defineConfig({
  test: {
    globals: true,      // no need to import describe/it/expect
    environment: 'jsdom',
    setupFiles: ['./src/test/setup.ts'],
    coverage: {
      provider: 'v8',
      reporter: ['text', 'lcov'],
      thresholds: { lines: 80, functions: 80 },
    },
  },
});

// ─── Testing pure functions ───────────────────────────────────────────────
// utils/cart.test.ts
import { calculateTotal, applyDiscount } from './cart';

describe('calculateTotal', () => {
  it('sums item prices times quantities', () => {
    const items = [
      { priceCents: 1000, qty: 2 },
      { priceCents: 500,  qty: 1 },
    ];
    expect(calculateTotal(items)).toBe(2500);
  });

  it('returns 0 for empty cart', () => {
    expect(calculateTotal([])).toBe(0);
  });
});

describe('applyDiscount', () => {
  it('applies percentage discount', () => {
    expect(applyDiscount(1000, { type: 'percent', value: 10 })).toBe(900);
  });

  it('does not go below 0', () => {
    expect(applyDiscount(100, { type: 'percent', value: 150 })).toBe(0);
  });
});
Concept 3

Testing Library — test what users see

React Testing Library's philosophy: test the UI the way users use it — find elements by role, text, and label; interact by clicking and typing; assert on visible state. Never test implementation details like state variable names or component internals.

// npm install -D @testing-library/react @testing-library/user-event
import { render, screen, waitFor } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { LoginForm } from './LoginForm';

describe('LoginForm', () => {
  it('shows error for empty submission', async () => {
    const user = userEvent.setup();
    render(<LoginForm onLogin={vi.fn()} />);

    await user.click(screen.getByRole('button', { name: 'Log In' }));

    expect(screen.getByText('Email is required')).toBeInTheDocument();
  });

  it('calls onLogin with credentials on valid submit', async () => {
    const user = userEvent.setup();
    const mockOnLogin = vi.fn().mockResolvedValue(undefined);
    render(<LoginForm onLogin={mockOnLogin} />);

    await user.type(screen.getByLabelText('Email'), 'alice@example.com');
    await user.type(screen.getByLabelText('Password'), 'secret123');
    await user.click(screen.getByRole('button', { name: 'Log In' }));

    await waitFor(() => {
      expect(mockOnLogin).toHaveBeenCalledWith({
        email: 'alice@example.com',
        password: 'secret123',
      });
    });
  });

  it('disables button while submitting', async () => {
    const user = userEvent.setup();
    let resolve;
    const pendingPromise = new Promise(r => { resolve = r; });
    render(<LoginForm onLogin={() => pendingPromise} />);

    await user.type(screen.getByLabelText('Email'), 'alice@example.com');
    await user.type(screen.getByLabelText('Password'), 'pw');
    await user.click(screen.getByRole('button', { name: 'Log In' }));

    expect(screen.getByRole('button', { name: 'Logging in...' })).toBeDisabled();
    resolve();
  });
});

// ─── Testing with mocked API calls (MSW) ─────────────────────────────────
import { setupServer } from 'msw/node';
import { http, HttpResponse } from 'msw';

const server = setupServer(
  http.get('/api/users/1', () => HttpResponse.json({ id: '1', name: 'Alice' })),
  http.get('/api/users/999', () => HttpResponse.json({ error: 'Not found' }, { status: 404 })),
);

beforeAll(() => server.listen());
afterEach(() => server.resetHandlers());
afterAll(() => server.close());

it('renders user profile from API', async () => {
  render(<UserProfile userId="1" />);
  expect(await screen.findByText('Alice')).toBeInTheDocument();
});
Queries in priority order (Testing Library guidance): getByRole → getByLabelText → getByPlaceholderText → getByText → getByTestId. If you're using getByTestId everywhere, you're testing implementation details. getByRole tests accessibility at the same time as functionality.

Concept 5

Error Boundaries — graceful degradation

Error boundaries catch JavaScript errors anywhere in their child component tree, preventing the whole app from crashing. They must be class components (no hook equivalent yet, though react-error-boundary wraps this nicely).

// npm install react-error-boundary
import { ErrorBoundary, useErrorBoundary } from 'react-error-boundary';

// ─── Using react-error-boundary ───────────────────────────────────────────
function ErrorFallback({ error, resetErrorBoundary }) {
  return (
    <div role="alert" className="error-boundary">
      <h2>Something went wrong</h2>
      <p>{error.message}</p>
      <button onClick={resetErrorBoundary}>Try again</button>
    </div>
  );
}

function App() {
  return (
    <ErrorBoundary
      FallbackComponent={ErrorFallback}
      onError={(error, info) => Sentry.captureException(error, { extra: info })}
      onReset={() => queryClient.clear()}
    >
      <Router />
    </ErrorBoundary>
  );
}

// ─── Granular boundaries — don't crash the whole page ────────────────────
function Dashboard() {
  return (
    <div>
      <DashboardHeader />  {/* critical — if this crashes, show full error */}

      <ErrorBoundary fallback={<p>Metrics unavailable.</p>}>
        <MetricsWidget />   {/* isolated — fallback if it crashes */}
      </ErrorBoundary>

      <ErrorBoundary fallback={<p>Activity feed unavailable.</p>}>
        <ActivityFeed />   {/* isolated — fallback if it crashes */}
      </ErrorBoundary>
    </div>
  );
}

// ─── Throwing errors to the boundary from hooks ───────────────────────────
function useRequiredData(id) {
  const { showBoundary } = useErrorBoundary();

  useEffect(() => {
    fetchData(id).catch(showBoundary);  // sends error to nearest boundary
  }, [id]);
}

Monitoring — Sentry and real user monitoring

// npm install @sentry/react
import * as Sentry from '@sentry/react';

// ─── Initialize (main.tsx) ────────────────────────────────────────────────
Sentry.init({
  dsn: import.meta.env.VITE_SENTRY_DSN,
  environment:  import.meta.env.VITE_ENV,
  release:      import.meta.env.VITE_GIT_SHA,
  tracesSampleRate: 0.1,        // 10% of transactions for performance monitoring
  replaysSessionSampleRate: 0.05,
  integrations: [
    Sentry.browserTracingIntegration(),
    Sentry.replayIntegration({ maskAllText: true }),
    Sentry.reactRouterV6BrowserTracingIntegration({ useEffect, useLocation, useNavigationType, createRoutesFromChildren, matchRoutes }),
  ],
});

// ─── Identify the user ────────────────────────────────────────────────────
function AppRoot() {
  const { user } = useAuth();

  useEffect(() => {
    if (user) {
      Sentry.setUser({ id: user.id, email: user.email });
    } else {
      Sentry.setUser(null);
    }
  }, [user]);

  return <App />;
}

// ─── Custom error boundary with Sentry ───────────────────────────────────
const SentryErrorBoundary = Sentry.withErrorBoundary(App, {
  fallback: <ErrorPage />,
  showDialog: false,
});

// ─── Manual error capture ─────────────────────────────────────────────────
async function handleCheckout(cartItems) {
  try {
    await processPayment(cartItems);
  } catch (err) {
    Sentry.captureException(err, {
      extra: { cartItemCount: cartItems.length, userId: user.id },
      tags:  { feature: 'checkout' },
    });
    showToast({ type: 'error', message: 'Payment failed. Our team has been notified.' });
  }
}

// ─── Web Vitals ───────────────────────────────────────────────────────────
import { onCLS, onINP, onLCP, onFCP, onTTFB } from 'web-vitals';

[onCLS, onINP, onLCP, onFCP, onTTFB].forEach(fn =>
  fn(({ name, value, rating }) =>
    analytics.track('web_vital', { name, value, rating })
  )
);
Never log PII to error monitoring tools. Sentry captures everything in scope on an exception — request bodies, component state, user data. Configure maskAllText in Sentry Replay, scrub sensitive fields in beforeSend, and never put tokens or passwords in component state.

Chapter 9 Takeaway

Production React requires four pillars: TypeScript for correctness at compile time, tests that verify user behaviour (not internals), Error Boundaries to contain failures gracefully, and monitoring to know what's broken before users report it. The CI/CD pipeline enforces all four automatically on every push — type-check, lint, test, build, deploy preview. This turns "it works on my machine" into "it works in production".
