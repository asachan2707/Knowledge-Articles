React Hooks

Hook 1

useState — local component state

import { useState } from 'react';

// Basic usage
const [value, setValue] = useState(initialValue);

// ─── Updating complex state ────────────────────────────────────────────────
// Always produce a new object/array — never mutate in place
function ProfileEditor({ initialUser }) {
  const [user, setUser] = useState(initialUser);

  const updateField = (field, value) =>
    setUser(prev => ({ ...prev, [field]: value }));

  const addTag = (tag) =>
    setUser(prev => ({ ...prev, tags: [...prev.tags, tag] }));

  const removeTag = (tag) =>
    setUser(prev => ({ ...prev, tags: prev.tags.filter(t => t !== tag) }));

  return (
    <form>
      <input value={user.name} onChange={e => updateField('name', e.target.value)} />
      <input value={user.email} onChange={e => updateField('email', e.target.value)} />
      {user.tags.map(tag => (
        <Badge key={tag} onRemove={() => removeTag(tag)}>{tag}</Badge>
      ))}
    </form>
  );
}

// ─── Multiple state slices vs one object ──────────────────────────────────
// Prefer multiple small state values when they change independently
const [firstName, setFirstName] = useState('');
const [lastName, setLastName]   = useState('');
const [age, setAge]             = useState(0);

// Group into an object when they always change together (form data)
const [formData, setFormData] = useState({ firstName: '', lastName: '', age: 0 });
Hook 2

useEffect — synchronizing with external systems

useEffect runs after render to synchronize your component with something outside React — APIs, subscriptions, timers, DOM measurements. The dependency array controls when it re-runs. The cleanup function prevents memory leaks.

import { useEffect, useState } from 'react';

// ─── Dependency array patterns ─────────────────────────────────────────────
useEffect(() => { /* runs after every render */ });
useEffect(() => { /* runs once on mount */ }, []);
useEffect(() => { /* runs when dep changes */ }, [dep]);

// ─── Data fetching with cleanup (abort on unmount/dependency change) ───────
function UserProfile({ userId }) {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const controller = new AbortController();
    setLoading(true);

    fetch(`/api/users/${userId}`, { signal: controller.signal })
      .then(res => res.json())
      .then(data => { setUser(data); setLoading(false); })
      .catch(err => { if (err.name !== 'AbortError') setLoading(false); });

    return () => controller.abort();  // cleanup: cancel in-flight request
  }, [userId]);  // re-run when userId changes

  if (loading) return <Spinner />;
  return <h1>{user?.name}</h1>;
}

// ─── Event listeners ──────────────────────────────────────────────────────
function WindowSize() {
  const [size, setSize] = useState({ width: window.innerWidth, height: window.innerHeight });

  useEffect(() => {
    const handleResize = () =>
      setSize({ width: window.innerWidth, height: window.innerHeight });

    window.addEventListener('resize', handleResize);
    return () => window.removeEventListener('resize', handleResize);  // cleanup
  }, []); // only once

  return <p>{size.width} × {size.height}</p>;
}

// ─── Subscriptions ─────────────────────────────────────────────────────────
function LivePrice({ symbol }) {
  const [price, setPrice] = useState(null);

  useEffect(() => {
    const ws = new WebSocket(`wss://prices.example.com/${symbol}`);
    ws.onmessage = (e) => setPrice(JSON.parse(e.data).price);
    return () => ws.close();  // cleanup: close WebSocket
  }, [symbol]);

  return <span>{price ? `$${price}` : 'Loading...'}</span>;
}

// ─── What NOT to put in useEffect ─────────────────────────────────────────
// Transforming data for rendering → do it during render (useMemo or plain)
// User event handlers → put directly in event handlers
// Derived state → compute from existing state, no useEffect needed
The most common useEffect mistake: missing a dependency in the array. The ESLint rule react-hooks/exhaustive-deps catches this. If you suppress the warning, you'll get stale closure bugs that are hard to debug. Fix the real issue instead: restructure so the function doesn't need the missing dep, or use useCallback.
Hook 3

useContext — consuming shared values without prop drilling

import { createContext, useContext, useState, useCallback } from 'react';

// ─── Create a context with a type-safe custom hook ─────────────────────────
const AuthContext = createContext(null);

export function AuthProvider({ children }) {
  const [user, setUser] = useState(null);

  const login = useCallback(async (email, password) => {
    const data = await apiLogin(email, password);
    setUser(data.user);
    localStorage.setItem('token', data.token);
  }, []);

  const logout = useCallback(() => {
    setUser(null);
    localStorage.removeItem('token');
  }, []);

  return (
    <AuthContext.Provider value={{ user, login, logout, isAuthenticated: !!user }}>
      {children}
    </AuthContext.Provider>
  );
}

// Custom hook with a clear error message if used outside provider
export function useAuth() {
  const ctx = useContext(AuthContext);
  if (!ctx) throw new Error('useAuth must be used inside AuthProvider');
  return ctx;
}

// ─── Wrap your app ────────────────────────────────────────────────────────
function App() {
  return (
    <AuthProvider>
      <Router>
        <Routes>{/* ... */}</Routes>
      </Router>
    </AuthProvider>
  );
}

// ─── Consume anywhere in the tree ─────────────────────────────────────────
function TopNav() {
  const { user, logout } = useAuth();
  return (
    <nav>
      {user ? <button onClick={logout}>Log out, {user.name}</button> : <LoginLink />}
    </nav>
  );
}

// ─── Performance: splitting contexts ─────────────────────────────────────
// If a context value changes frequently, split into stable and dynamic parts
// to avoid re-rendering all consumers on every change
const UserContext   = createContext(null);  // changes on login/logout only
const ThemeContext  = createContext(null);  // changes on theme toggle
// Consumers of UserContext don't re-render when theme changes
Hook 4

useReducer — complex state with actions

useReducer is useState for when state logic gets complex. It is Redux-style — you dispatch actions, a pure reducer function computes the next state. Best when: multiple sub-values, or next state depends on the previous in complex ways.

import { useReducer } from 'react';

// ─── Real example: shopping cart ─────────────────────────────────────────
const cartReducer = (state, action) => {
  switch (action.type) {
    case 'ADD_ITEM': {
      const existing = state.items.find(i => i.id === action.product.id);
      if (existing) {
        return {
          ...state,
          items: state.items.map(i =>
            i.id === action.product.id ? { ...i, qty: i.qty + 1 } : i
          ),
        };
      }
      return { ...state, items: [...state.items, { ...action.product, qty: 1 }] };
    }
    case 'REMOVE_ITEM':
      return { ...state, items: state.items.filter(i => i.id !== action.id) };

    case 'UPDATE_QTY':
      return {
        ...state,
        items: state.items.map(i =>
          i.id === action.id ? { ...i, qty: Math.max(1, action.qty) } : i
        ),
      };

    case 'CLEAR':
      return { ...state, items: [] };

    default:
      return state;
  }
};

const initialCart = { items: [], coupon: null };

function Cart() {
  const [cart, dispatch] = useReducer(cartReducer, initialCart);

  const total = cart.items.reduce((sum, i) => sum + i.priceCents * i.qty, 0);

  return (
    <div>
      {cart.items.map(item => (
        <CartRow
          key={item.id}
          item={item}
          onRemove={() => dispatch({ type: 'REMOVE_ITEM', id: item.id })}
          onQtyChange={qty => dispatch({ type: 'UPDATE_QTY', id: item.id, qty })}
        />
      ))}
      <p>Total: ${(total / 100).toFixed(2)}</p>
      <button onClick={() => dispatch({ type: 'CLEAR' })}>Clear cart</button>
    </div>
  );
}
When to use useReducer vs useState: useState is simpler and usually sufficient. Switch to useReducer when state has more than 3–4 sub-values that transition together, when the same state update logic is called from multiple places, or when you want to write pure tests for the state logic in isolation.
Hook 5

useRef — DOM access and mutable values that don't trigger re-renders

import { useRef, useEffect, useState } from 'react';

// ─── DOM access ──────────────────────────────────────────────────────────
function AutoFocusInput() {
  const inputRef = useRef(null);

  useEffect(() => {
    inputRef.current?.focus();  // focus on mount
  }, []);

  return <input ref={inputRef} placeholder="Auto-focused" />;
}

// ─── Video player controls ─────────────────────────────────────────────
function VideoPlayer({ src }) {
  const videoRef = useRef(null);

  const play  = () => videoRef.current?.play();
  const pause = () => videoRef.current?.pause();

  return (
    <div>
      <video ref={videoRef} src={src} />
      <button onClick={play}>Play</button>
      <button onClick={pause}>Pause</button>
    </div>
  );
}

// ─── Storing mutable values without triggering re-render ─────────────────
// Classic use case: tracking the previous value of a prop/state
function Counter() {
  const [count, setCount] = useState(0);
  const prevCountRef = useRef(0);

  useEffect(() => {
    prevCountRef.current = count;  // update after render, no re-render triggered
  });

  return (
    <p>
      Now: {count} | Before: {prevCountRef.current}
      <button onClick={() => setCount(c => c + 1)}>+</button>
    </p>
  );
}

// ─── Storing timer/interval IDs ────────────────────────────────────────
function Stopwatch() {
  const [elapsed, setElapsed] = useState(0);
  const intervalRef = useRef(null);

  const start = () => {
    if (intervalRef.current) return; // already running
    intervalRef.current = setInterval(() => setElapsed(e => e + 1), 1000);
  };

  const stop = () => {
    clearInterval(intervalRef.current);
    intervalRef.current = null;
  };

  useEffect(() => () => clearInterval(intervalRef.current), []); // cleanup on unmount

  return (
    <div>
      <p>{elapsed}s</p>
      <button onClick={start}>Start</button>
      <button onClick={stop}>Stop</button>
    </div>
  );
}
Hook 6

Custom Hooks — extracting and sharing stateful logic

A custom hook is a function that starts with use and calls other hooks. They are the best way to share stateful logic between components — far better than HOCs or render props for this purpose.

// ─── useFetch — generic data fetching ─────────────────────────────────────
function useFetch(url) {
  const [data, setData]     = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError]   = useState(null);

  useEffect(() => {
    if (!url) return;
    const controller = new AbortController();
    setLoading(true);
    setError(null);

    fetch(url, { signal: controller.signal })
      .then(res => { if (!res.ok) throw new Error(`HTTP ${res.status}`); return res.json(); })
      .then(data => { setData(data); setLoading(false); })
      .catch(err => { if (err.name !== 'AbortError') { setError(err.message); setLoading(false); } });

    return () => controller.abort();
  }, [url]);

  return { data, loading, error };
}

// Usage
function Products() {
  const { data: products, loading, error } = useFetch('/api/products');
  if (loading) return <Spinner />;
  if (error)   return <Error message={error} />;
  return <ProductGrid products={products} />;
}

// ─── useLocalStorage — persistent state ──────────────────────────────────
function useLocalStorage(key, initialValue) {
  const [value, setValue] = useState(() => {
    try { return JSON.parse(localStorage.getItem(key)) ?? initialValue; }
    catch { return initialValue; }
  });

  const setStoredValue = (val) => {
    const nextVal = typeof val === 'function' ? val(value) : val;
    setValue(nextVal);
    localStorage.setItem(key, JSON.stringify(nextVal));
  };

  return [value, setStoredValue];
}

// Usage — drop-in replacement for useState
const [theme, setTheme] = useLocalStorage('theme', 'light');

// ─── useDebounce — delay expensive operations ─────────────────────────────
function useDebounce(value, delay = 300) {
  const [debounced, setDebounced] = useState(value);

  useEffect(() => {
    const timer = setTimeout(() => setDebounced(value), delay);
    return () => clearTimeout(timer);
  }, [value, delay]);

  return debounced;
}

// Usage — delay search query before firing API call
function SearchPage() {
  const [query, setQuery] = useState('');
  const debouncedQuery = useDebounce(query, 400);

  const { data: results } = useFetch(
    debouncedQuery ? `/api/search?q=${debouncedQuery}` : null
  );

  return (
    <div>
      <input value={query} onChange={e => setQuery(e.target.value)} />
      {results?.map(r => <SearchResult key={r.id} result={r} />)}
    </div>
  );
}

// ─── useOnClickOutside — close dropdowns on outside click ─────────────────
function useOnClickOutside(ref, handler) {
  useEffect(() => {
    const listener = (e) => {
      if (!ref.current || ref.current.contains(e.target)) return;
      handler(e);
    };
    document.addEventListener('mousedown', listener);
    return () => document.removeEventListener('mousedown', listener);
  }, [ref, handler]);
}

function Dropdown({ children, content }) {
  const [open, setOpen] = useState(false);
  const ref = useRef(null);
  useOnClickOutside(ref, () => setOpen(false));

  return (
    <div ref={ref}>
      <button onClick={() => setOpen(o => !o)}>{children}</button>
      {open && <div className="dropdown-content">{content}</div>}
    </div>
  );
}
Name custom hooks with use prefix — this is not just convention, it is required. React's linter uses it to know that hook rules apply. A function called useAuth gets hook rule checking; a function called getAuth does not, even if it calls hooks inside.

Chapter 4 Takeaway

Hooks replace the need for class components entirely. useState for local state, useEffect for synchronizing with external systems (with cleanup), useContext for shared values, useReducer for complex state machines, and useRef for DOM access or mutable values that shouldn't trigger renders. Custom hooks are how you share all this logic — extract early, name clearly, keep focused on one concern.
