Concept 1

Next.js — the React framework for production

Next.js adds what React doesn't provide out of the box: routing, server-side rendering, static generation, image optimization, and API routes. The App Router (Next 13+) is the modern architecture built on React Server Components.

// ─── App Router file structure ────────────────────────────────────────────
// app/
//   layout.tsx          ← root layout (persists across navigations)
//   page.tsx            ← home page /
//   products/
//     page.tsx          ← /products
//     [productId]/
//       page.tsx        ← /products/:productId
//       loading.tsx     ← loading UI (automatic Suspense)
//       error.tsx       ← error boundary
//   dashboard/
//     layout.tsx        ← dashboard-specific layout

// ─── Root layout ──────────────────────────────────────────────────────────
// app/layout.tsx
export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <body>
        <TopNav />
        {children}
        <Footer />
      </body>
    </html>
  );
}

// ─── Server Component page — fetches data on the server ──────────────────
// app/products/[productId]/page.tsx
interface PageProps {
  params: Promise<{ productId: string }>;  // Next 15: params is a Promise
}

export default async function ProductPage({ params }: PageProps) {
  const { productId } = await params;

  // Runs on the server — no client JS, no loading state visible
  const product = await db.product.findUnique({ where: { id: productId } });

  if (!product) notFound();  // from 'next/navigation'

  return (
    <div>
      <h1>{product.name}</h1>
      <p>${(product.priceCents / 100).toFixed(2)}</p>
      <AddToCartButton productId={productId} /> {/* client component */}
    </div>
  );
}

// SEO metadata — static or dynamic
export async function generateMetadata({ params }: PageProps) {
  const { productId } = await params;
  const product = await db.product.findUnique({ where: { id: productId } });
  return { title: product?.name ?? 'Product', description: product?.description };
}

// ─── Client Component — interactive parts ─────────────────────────────────
// 'use client' marks this as a client component
'use client';
function AddToCartButton({ productId }: { productId: string }) {
  const addItem = useCartStore(s => s.addItem);
  return <button onClick={() => addItem(productId)}>Add to Cart</button>;
}
Concept 2

Vite — the build tool that replaced Create React App

Vite uses native ES modules in development for instant start (<300ms regardless of project size) and Rollup for production builds. It replaced CRA as the standard for client-side React apps.

// ─── Create a Vite + React + TypeScript project ───────────────────────────
// npm create vite@latest my-app -- --template react-ts

// vite.config.ts — typical production config
import { defineConfig } from 'vite';
import react from '@vitejs/plugin-react';
import { visualizer } from 'rollup-plugin-visualizer';
import path from 'path';

export default defineConfig({
  plugins: [
    react({
      babel: { plugins: [['babel-plugin-react-compiler', {}]] }, // React Compiler
    }),
    visualizer({ open: false }),
  ],
  resolve: {
    alias: { '@': path.resolve(__dirname, './src') },  // @/components/Button
  },
  build: {
    rollupOptions: {
      output: {
        manualChunks: {
          vendor:  ['react', 'react-dom'],
          router:  ['react-router-dom'],
          query:   ['@tanstack/react-query'],
        },
      },
    },
  },
  test: {
    globals: true,
    environment: 'jsdom',
    setupFiles: ['./src/test/setup.ts'],
  },
});

// ─── Environment variables ────────────────────────────────────────────────
// .env.local (gitignored)
VITE_API_URL=http://localhost:3000
VITE_SENTRY_DSN=...

// Usage in code — only VITE_ prefixed vars are exposed to client
const apiUrl = import.meta.env.VITE_API_URL;
Concept 3

Tailwind CSS — utility-first styling

Tailwind provides low-level utility classes that you compose directly in JSX. No context switching between CSS files and components, no naming things, no specificity battles. The final CSS bundle is tiny because only used classes are included.

// ─── Tailwind patterns in React ──────────────────────────────────────────

// Simple card
function ProductCard({ product }) {
  return (
    <div className="rounded-xl border border-gray-200 bg-white shadow-sm overflow-hidden hover:shadow-md transition-shadow">
      <img src={product.imageUrl} alt={product.name} className="w-full h-48 object-cover" />
      <div className="p-4">
        <h3 className="font-semibold text-gray-900 text-lg">{product.name}</h3>
        <p className="text-gray-500 text-sm mt-1">{product.description}</p>
        <p className="text-blue-600 font-bold text-xl mt-3">
          ${(product.priceCents / 100).toFixed(2)}
        </p>
        <button className="mt-4 w-full bg-blue-600 hover:bg-blue-700 text-white font-semibold py-2 px-4 rounded-lg transition-colors">
          Add to Cart
        </button>
      </div>
    </div>
  );
}

// ─── Conditional classes with clsx/cn ────────────────────────────────────
// npm install clsx tailwind-merge
import { clsx } from 'clsx';
import { twMerge } from 'tailwind-merge';

const cn = (...inputs) => twMerge(clsx(inputs));

function Button({ variant, size, className, children, ...props }) {
  return (
    <button
      className={cn(
        'inline-flex items-center justify-center font-semibold rounded-lg transition-colors',
        {
          'bg-blue-600 hover:bg-blue-700 text-white':      variant === 'primary',
          'bg-gray-100 hover:bg-gray-200 text-gray-700':   variant === 'secondary',
          'bg-red-600 hover:bg-red-700 text-white':        variant === 'danger',
          'px-3 py-1.5 text-sm':                           size === 'sm',
          'px-4 py-2 text-base':                           size === 'md',
          'px-6 py-3 text-lg':                             size === 'lg',
        },
        className  // allow callers to extend
      )}
      {...props}
    >
      {children}
    </button>
  );
}
Concept 4

shadcn/ui — components you own, not install

shadcn/ui is not a traditional component library — it generates component source code into your project. You own the code, customize freely, and there is no version lock-in. Built on Radix UI (accessible primitives) + Tailwind.

# Add components to your project (they become source files you own)
npx shadcn@latest init
npx shadcn@latest add button input dialog select table

// The components live in src/components/ui/ — modify them freely
import { Button } from '@/components/ui/button';
import { Dialog, DialogContent, DialogHeader, DialogTitle } from '@/components/ui/dialog';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select';

function InviteUserDialog({ open, onOpenChange }) {
  const [email, setEmail] = useState('');
  const [role, setRole]   = useState('viewer');

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="sm:max-w-md">
        <DialogHeader>
          <DialogTitle>Invite team member</DialogTitle>
        </DialogHeader>

        <div className="grid gap-4 py-4">
          <div className="grid gap-2">
            <Label htmlFor="email">Email</Label>
            <Input
              id="email"
              type="email"
              value={email}
              onChange={e => setEmail(e.target.value)}
              placeholder="colleague@company.com"
            />
          </div>

          <div className="grid gap-2">
            <Label>Role</Label>
            <Select value={role} onValueChange={setRole}>
              <SelectTrigger><SelectValue /></SelectTrigger>
              <SelectContent>
                <SelectItem value="admin">Admin</SelectItem>
                <SelectItem value="editor">Editor</SelectItem>
                <SelectItem value="viewer">Viewer</SelectItem>
              </SelectContent>
            </Select>
          </div>
        </div>

        <div className="flex justify-end gap-2">
          <Button variant="outline" onClick={() => onOpenChange(false)}>Cancel</Button>
          <Button onClick={() => sendInvite(email, role)}>Send Invite</Button>
        </div>
      </DialogContent>
    </Dialog>
  );
}
Concept 5

Motion — declarative animations

Motion (formerly Framer Motion) is the standard animation library for React. Declarative API — you describe the animation state, Motion handles the interpolation. Works with React's lifecycle and layout changes.

// npm install motion
import { motion, AnimatePresence } from 'motion/react';

// ─── Basic animation ──────────────────────────────────────────────────────
function FadeInCard({ children }) {
  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.3, ease: 'easeOut' }}
    >
      {children}
    </motion.div>
  );
}

// ─── AnimatePresence — animate on unmount ────────────────────────────────
function Toast({ toasts }) {
  return (
    <div className="toast-container">
      <AnimatePresence>
        {toasts.map(toast => (
          <motion.div
            key={toast.id}
            initial={{ opacity: 0, x: 100 }}
            animate={{ opacity: 1, x: 0 }}
            exit={{ opacity: 0, x: 100 }}
            transition={{ type: 'spring', stiffness: 300, damping: 30 }}
            className="toast"
          >
            {toast.message}
          </motion.div>
        ))}
      </AnimatePresence>
    </div>
  );
}

// ─── Layout animations — animate DOM position changes ─────────────────────
function SortableList({ items }) {
  return (
    <ul>
      {items.map(item => (
        <motion.li key={item.id} layout transition={{ type: 'spring', stiffness: 400 }}>
          {item.name}
        </motion.li>
      ))}
    </ul>
  );
  // When items reorder, Motion smoothly animates each item to its new position
}

// ─── Gesture-driven animation ─────────────────────────────────────────────
function DraggableCard({ children }) {
  return (
    <motion.div
      drag
      dragConstraints={{ left: 0, right: 300, top: 0, bottom: 300 }}
      whileDrag={{ scale: 1.05, cursor: 'grabbing' }}
      whileHover={{ scale: 1.02 }}
      whileTap={{ scale: 0.98 }}
      className="card"
    >
      {children}
    </motion.div>
  );
}
Concept 6

Storybook — isolated component development

Storybook renders components in isolation — no routing, no auth, no global state. Each "story" is one visual state of a component. It doubles as a living component library for designers and developers.

// npx storybook@latest init

// Button.stories.tsx
import type { Meta, StoryObj } from '@storybook/react';
import { Button } from './Button';

const meta: Meta<typeof Button> = {
  component: Button,
  title: 'UI/Button',
  tags: ['autodocs'],   // generates documentation page automatically
  argTypes: {
    variant: { control: 'select', options: ['primary', 'secondary', 'danger', 'ghost'] },
    size:    { control: 'select', options: ['sm', 'md', 'lg'] },
    loading: { control: 'boolean' },
  },
};

export default meta;
type Story = StoryObj<typeof Button>;

export const Primary: Story = {
  args: { children: 'Click me', variant: 'primary' },
};

export const Loading: Story = {
  args: { children: 'Saving...', variant: 'primary', loading: true },
};

export const AllVariants: Story = {
  render: () => (
    <div style={{ display: 'flex', gap: '8px' }}>
      <Button variant="primary">Primary</Button>
      <Button variant="secondary">Secondary</Button>
      <Button variant="danger">Danger</Button>
      <Button variant="ghost">Ghost</Button>
    </div>
  ),
};

// ─── Testing in Storybook (Storybook Test) ─────────────────────────────────
// Stories become test cases in Vitest
import { composeStories } from '@storybook/react';
import * as stories from './Button.stories';

const { Primary, Loading } = composeStories(stories);

test('Primary button renders', async () => {
  await render(<Primary />);
  expect(screen.getByRole('button', { name: 'Click me' })).toBeInTheDocument();
});
Storybook as the component contract: Write stories before implementing, collaborate with design, catch regressions with visual testing (Chromatic). The discipline of writing stories forces you to think about the full range of states a component needs to handle.

Chapter 10 Takeaway

The 2026 React stack: Vite for fast local development, Next.js when you need SSR/RSC, Tailwind for styling, shadcn/ui for components you own and customize, Motion for animations, and Storybook for isolated component development. These tools are additive — each solves a specific problem without fighting the others. Start with Vite + Tailwind + shadcn/ui for most apps; add Next.js when SEO or SSR matters.
