Concept 1

Fetch API — the browser's built-in HTTP client

// ─── Creating a typed API client ──────────────────────────────────────────
class ApiError extends Error {
  constructor(status, message, body) {
    super(message);
    this.status = status;
    this.body = body;
  }
}

async function apiFetch(path, options = {}) {
  const token = localStorage.getItem('token');

  const res = await fetch(`${import.meta.env.VITE_API_URL}${path}`, {
    headers: {
      'Content-Type': 'application/json',
      ...(token ? { Authorization: `Bearer ${token}` } : {}),
      ...options.headers,
    },
    ...options,
    body: options.body ? JSON.stringify(options.body) : undefined,
  });

  if (!res.ok) {
    const body = await res.json().catch(() => ({}));
    throw new ApiError(res.status, body.message ?? `HTTP ${res.status}`, body);
  }

  if (res.status === 204) return null;
  return res.json();
}

// ─── Usage ────────────────────────────────────────────────────────────────
const users    = await apiFetch('/users');
const user     = await apiFetch('/users/123');
const newUser  = await apiFetch('/users', { method: 'POST', body: { name, email } });
const updated  = await apiFetch('/users/123', { method: 'PATCH', body: { name } });
await apiFetch('/users/123', { method: 'DELETE' });

Axios — batteries-included HTTP client

Axios is popular in older React codebases. Key differences from fetch: automatic JSON parse, error on non-2xx status, interceptors for auth/logging, request cancellation, upload progress.

// npm install axios
import axios from 'axios';

// ─── Axios instance with interceptors ────────────────────────────────────
const api = axios.create({
  baseURL: import.meta.env.VITE_API_URL,
  timeout: 10000,
});

// Request interceptor — add auth header
api.interceptors.request.use(config => {
  const token = localStorage.getItem('token');
  if (token) config.headers.Authorization = `Bearer ${token}`;
  return config;
});

// Response interceptor — global error handling + token refresh
api.interceptors.response.use(
  res => res.data,   // unwrap .data automatically
  async err => {
    if (err.response?.status === 401 && !err.config._retry) {
      err.config._retry = true;
      const newToken = await refreshToken();
      localStorage.setItem('token', newToken);
      err.config.headers.Authorization = `Bearer ${newToken}`;
      return api(err.config);  // retry original request
    }
    return Promise.reject(err);
  }
);

// Usage — clean because interceptor unwraps .data
const users = await api.get('/users');             // already the array
const user  = await api.post('/users', { name });  // already the created object

// Upload with progress
const upload = async (file) => {
  return api.post('/uploads', formData, {
    headers: { 'Content-Type': 'multipart/form-data' },
    onUploadProgress: e => setProgress(Math.round((e.loaded / e.total) * 100)),
  });
};
Concept 3

TanStack Query — server state synchronization

TanStack Query manages the full lifecycle of server data: fetching, caching, background revalidation, retries, and mutations with rollback. It eliminates the loading/error/data state you would otherwise manage manually.

// npm install @tanstack/react-query
import { QueryClient, QueryClientProvider, useQuery, useMutation, useQueryClient } from '@tanstack/react-query';

const queryClient = new QueryClient({
  defaultOptions: {
    queries: { staleTime: 5 * 60 * 1000, retry: 2 },
  },
});

function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <Router />
    </QueryClientProvider>
  );
}

// ─── useQuery — fetch and cache data ──────────────────────────────────────
function UserProfile({ userId }) {
  const { data: user, isLoading, error, isFetching } = useQuery({
    queryKey: ['user', userId],      // cache key — changing userId auto-refetches
    queryFn: () => apiFetch(`/users/${userId}`),
    staleTime: 5 * 60 * 1000,       // 5 min before refetch
    gcTime: 10 * 60 * 1000,         // 10 min before garbage collect
    enabled: !!userId,              // don't fetch if userId is null
  });

  if (isLoading) return <Skeleton />;
  if (error)     return <Error message={error.message} />;

  return (
    <div>
      <h1>{user.name}</h1>
      {isFetching && <RefreshSpinner />} {/* background refetch indicator */}
    </div>
  );
}

// ─── useMutation — modify server data ─────────────────────────────────────
function UpdateUserForm({ userId }) {
  const queryClient = useQueryClient();

  const mutation = useMutation({
    mutationFn: (data) => apiFetch(`/users/${userId}`, { method: 'PATCH', body: data }),
    onSuccess: (updatedUser) => {
      // Update cache directly — no refetch needed
      queryClient.setQueryData(['user', userId], updatedUser);
      // Or invalidate to trigger a refetch:
      // queryClient.invalidateQueries({ queryKey: ['user', userId] });
    },
    onError: (error) => showToast({ type: 'error', message: error.message }),
  });

  const handleSubmit = (formData) => mutation.mutate(formData);

  return (
    <form onSubmit={handleSubmit}>
      {/* form fields */}
      <Button type="submit" loading={mutation.isPending}>Save</Button>
    </form>
  );
}

Concept 4

SWR — stale-while-revalidate from Vercel

SWR is the Vercel-maintained alternative to TanStack Query. Simpler API, slightly smaller bundle. Best when you need automatic revalidation on focus/reconnect and don't need advanced mutation patterns.

// npm install swr
import useSWR, { useSWRConfig } from 'swr';
import useSWRMutation from 'swr/mutation';

// Global fetcher config
const fetcher = (url) => fetch(url).then(r => r.json());

// ─── Basic usage ──────────────────────────────────────────────────────────
function UserCard({ userId }) {
  const { data: user, error, isLoading, isValidating } = useSWR(
    userId ? `/api/users/${userId}` : null,  // null key = don't fetch
    fetcher,
    {
      revalidateOnFocus: true,   // refetch when tab regains focus
      revalidateOnReconnect: true,
      refreshInterval: 0,        // 0 = no polling (set >0 for live data)
    }
  );

  if (isLoading) return <Skeleton />;
  if (error)     return <Error />;
  return <div>{user.name} {isValidating && '⟳'}</div>;
}

// ─── Mutations with SWR ───────────────────────────────────────────────────
const updateUser = async (url, { arg }) =>
  fetch(url, { method: 'PATCH', body: JSON.stringify(arg), headers: {'Content-Type':'application/json'} }).then(r => r.json());

function EditUser({ userId }) {
  const { trigger, isMutating } = useSWRMutation(`/api/users/${userId}`, updateUser);

  const handleSave = async (data) => {
    await trigger(data);
    // SWR automatically updates the cache for this key
  };

  return <Button onClick={() => handleSave({ name: 'New Name' }) } loading={isMutating}>Save</Button>;
}
Concept 5

WebSockets — real-time data in React

// ─── Custom hook for WebSocket connection ─────────────────────────────────
function useWebSocket(url) {
  const [messages, setMessages]   = useState([]);
  const [status, setStatus]       = useState('connecting');
  const wsRef = useRef(null);

  useEffect(() => {
    if (!url) return;

    const ws = new WebSocket(url);
    wsRef.current = ws;

    ws.onopen    = () => setStatus('connected');
    ws.onclose   = () => setStatus('disconnected');
    ws.onerror   = () => setStatus('error');
    ws.onmessage = (e) => {
      const msg = JSON.parse(e.data);
      setMessages(prev => [...prev, msg]);
    };

    return () => ws.close();
  }, [url]);

  const send = useCallback((data) => {
    if (wsRef.current?.readyState === WebSocket.OPEN) {
      wsRef.current.send(JSON.stringify(data));
    }
  }, []);

  return { messages, status, send };
}

// ─── Live order tracking ──────────────────────────────────────────────────
function OrderTracker({ orderId }) {
  const token = useAuthToken();
  const { messages, status } = useWebSocket(
    orderId ? `wss://api.example.com/orders/${orderId}/track?token=${token}` : null
  );

  const latest = messages[messages.length - 1];

  return (
    <div>
      <ConnectionBadge status={status} />
      {latest && <OrderStatusCard status={latest.status} location={latest.location} />}
      <Timeline events={messages} />
    </div>
  );
}

// ─── Combining WebSocket with TanStack Query ──────────────────────────────
// WebSocket updates the query cache directly — query data stays fresh in real time
function useLiveOrderStatus(orderId) {
  const queryClient = useQueryClient();

  // Initial fetch via query
  const query = useQuery({
    queryKey: ['order', orderId],
    queryFn: () => apiFetch(`/orders/${orderId}`),
  });

  // WebSocket updates
  useEffect(() => {
    if (!orderId) return;
    const ws = new WebSocket(`wss://api.example.com/orders/${orderId}/live`);
    ws.onmessage = (e) => {
      const update = JSON.parse(e.data);
      // Patch query cache in place
      queryClient.setQueryData(['order', orderId], prev => ({ ...prev, ...update }));
    };
    return () => ws.close();
  }, [orderId, queryClient]);

  return query;
}
Concept 6

Optimistic Updates — instant UI feedback

Optimistic updates assume a mutation will succeed and update the UI immediately — then roll back if the server returns an error. This makes interactions feel instant even over slow networks.

// ─── TanStack Query optimistic mutation ───────────────────────────────────
function TodoList({ listId }) {
  const queryClient = useQueryClient();

  const toggleTodo = useMutation({
    mutationFn: ({ id, done }) =>
      apiFetch(`/todos/${id}`, { method: 'PATCH', body: { done } }),

    onMutate: async ({ id, done }) => {
      // Cancel any in-flight refetches
      await queryClient.cancelQueries({ queryKey: ['todos', listId] });

      // Snapshot previous value
      const previous = queryClient.getQueryData(['todos', listId]);

      // Optimistically update cache
      queryClient.setQueryData(['todos', listId], (old) =>
        old.map(todo => todo.id === id ? { ...todo, done } : todo)
      );

      return { previous };  // context for rollback
    },

    onError: (err, variables, context) => {
      // Rollback on error
      queryClient.setQueryData(['todos', listId], context.previous);
      showToast({ type: 'error', message: 'Failed to update. Changes reverted.' });
    },

    onSettled: () => {
      // Always refetch after error or success to stay in sync with server
      queryClient.invalidateQueries({ queryKey: ['todos', listId] });
    },
  });

  const { data: todos } = useQuery({
    queryKey: ['todos', listId],
    queryFn: () => apiFetch(`/lists/${listId}/todos`),
  });

  return (
    <ul>
      {todos?.map(todo => (
        <li
          key={todo.id}
          style={{ opacity: todo.done ? 0.5 : 1 }}
          onClick={() => toggleTodo.mutate({ id: todo.id, done: !todo.done })}
        >
          {todo.text}
        </li>
      ))}
    </ul>
  );
}

// ─── Simple optimistic pattern without TanStack Query ────────────────────
function LikeButton({ postId, initialLiked }) {
  const [liked, setLiked] = useState(initialLiked);
  const [count, setCount] = useState(0);

  const handleLike = async () => {
    const nextLiked = !liked;
    setLiked(nextLiked);     // optimistic
    setCount(c => c + (nextLiked ? 1 : -1));

    try {
      await apiFetch(`/posts/${postId}/like`, {
        method: nextLiked ? 'POST' : 'DELETE',
      });
    } catch {
      setLiked(liked);       // rollback
      setCount(c => c + (liked ? 1 : -1));
    }
  };

  return (
    <button onClick={handleLike} aria-pressed={liked}>
      {liked ? '❤️' : '🤍'} {count}
    </button>
  );
}
The three phases of an optimistic update: (1) snapshot the current cache, (2) immediately apply the expected change, (3) on error, roll back to the snapshot. Always invalidate the query on settlement to ensure eventual consistency with the server.

Chapter 7 Takeaway

Build a typed fetch wrapper once and use it everywhere. Use TanStack Query or SWR for all server data — they replace the loading/error/data tri-state pattern you would otherwise repeat on every page. WebSockets and TanStack Query work beautifully together: Query handles initial fetch and cache management, WebSocket pushes incremental updates directly into the cache. Optimistic updates make the UI feel fast; just always have a rollback path.
